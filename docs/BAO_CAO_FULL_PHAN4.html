                                                                                                                                                                                                                            C. Công thức tổng hợp

                                                                                                                                                                                                                            1. Động cơ DC:

                                                                                                                                                                                                                            U = R_a × I_a + L_a × dI_a/dt + K_e × ω
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            E_a = K_e × ω
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            M_em = K_m × I_a
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            J × dω/dt = M_em - M_load - B × ω
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            n = (U / K_e) - (R_a / K_e²) × M
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            T_a = L_a / R_a
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            T_m = (J × R_a) / K_e²
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            2. PWM:
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            U_avg = D × V_supply
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            ΔI ≈ (V - E_a) × D × (1-D) / (L_a × f_PWM)
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            P_loss_MOSFET = I² × R_ds_on + V × I × t_sw × f_PWM
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            η_ESC = P_out / (P_out + P_loss)
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            3. Hiệu suất:
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            P_out = M × ω
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            P_in = U × I
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            η = P_out / P_in
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            P_loss = P_Cu + P_cơ + P_sắt + P_ESC
                                                                                                                                                                                                                            

                                                                                                                                                                                                                            P_Cu = I² × R_a
